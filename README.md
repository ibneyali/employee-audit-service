# employee-audit-service

Having below schema

EMPLOYEE {
	ID (PK)
	FIRST_NAME
	LAST_NAME
	EMAIL
	PHONE
	DEPARTMENT_ID (FK → DEPARTMENT.ID)
	ADDRESS_ID (FK → ADDRESS.ID)
	CREATED_TIMESTAMP
	UPDATED_TIMESTAMP
	UPDATED_BY
}
ADDRESS {
	ID (PK)
	ADDRESS_LINE_1
	ADDRESS_LINE_2
	ADDRESS_LINE_3
	COUNTRY
	POSTAL_CODE
	CREATED_TIMESTAMP
	UPDATED_TIMESTAMP
	UPDATED_BY
}
DEPARTMENT {
	ID (PK)
	NAME
	CREATED_TIMESTAMP
	UPDATED_TIMESTAMP
	UPDATED_BY

}
TRAINING {
	ID (PK)
	NAME
	DESCRIPTION
	CREATED_TIMESTAMP
	UPDATED_TIMESTAMP
	UPDATED_BY
}
EMPLOYEE_TRAINING {
	TRAINING_ID (FK → TRAINING.ID)
	EMP_ID (FK → EMPLOYEE.ID)
	DATE_OF_ALLOCATION
	STATUS
	CREATED_TIMESTAMP
	UPDATED_TIMESTAMP
	UPDATED_BY
}
